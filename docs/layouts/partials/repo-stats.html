{{- $repo := "elC0mpa/aws-doctor" -}} {{- $repoUrl := printf "https://api.github.com/repos/%s" $repo -}} {{- $opts := dict "headers" (dict "User-Agent" "Hugo-Hextra") -}} {{- /* Fetch Repo Data (Forks) */ -}} {{- $forks := 0 -}} {{- $repoRes := resources.GetRemote $repoUrl $opts -}} {{- if $repoRes -}} {{- $data := $repoRes | transform.Unmarshal -}} {{- if $data -}} {{- $forks = index $data "forks_count" | default 0 -}} {{- end -}} {{- end -}} {{- /* Fetch Contributors Count */ -}} {{- $contributors := 1 -}} {{- $contribUrl := printf "https://api.github.com/repos/%s/contributors" $repo -}} {{- $contribRes := resources.GetRemote $contribUrl $opts -}} {{- if $contribRes -}} {{- $data := $contribRes | transform.Unmarshal -}} {{- if reflect.IsSlice $data -}} {{- $contributors = len $data -}} {{- end -}} {{- end -}} {{- /* Labels */ -}} {{- $contribLabel := .contribLabel | default "Contributors" -}} {{- $forksLabel := .forksLabel | default "Forks" -}}
{{ $contributors }} {{ $contribLabel }}
{{ $forks }} {{ $forksLabel }}
